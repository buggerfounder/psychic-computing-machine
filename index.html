Пример кода
local TweenService = game:GetService("TweenService")

local function playHitEffect(part)
  local tweenInfo = TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
  local goal = {Transparency = 0.5, Color = Color3.fromRGB(255, 0, 0)}
  local tween = TweenService:Create(part, tweenInfo, goal)
  tween:Play()
  tween.Completed:Wait()
  goal = {Transparency = 0, Color = Color3.fromRGB(255, 255, 255)}
  tween = TweenService:Create(part, tweenInfo, goal)
  tween:Play()
end
Эффект хита с изменением цвета и прозрачности
